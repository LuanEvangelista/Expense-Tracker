using ExpenseTracker.Services;var service = new ExpenseService();if (args.Length == 0){    Console.WriteLine("No command provided.");    return;}var command = args[0].ToLower();string GetArgValue(string flag){    int index = Array.IndexOf(args, flag);    return (index != -1 && index + 1 < args.Length) ? args[index + 1] : string.Empty;}switch (command){    case "add":        var desc = GetArgValue("--description");        var amountStr = GetArgValue("--amount");        if (decimal.TryParse(amountStr, out decimal amount))            service.AddExpense(desc, amount);        break;    case "list":        service.ListExpenses();        break;    case "summary":        var monthStr = GetArgValue("--month");        if (int.TryParse(monthStr, out int month))            service.ShowSummary(month);        else            service.ShowSummary();        break;    case "delete":        var idStr = GetArgValue("--id");        if (int.TryParse(idStr, out int id))            service.DeleteExpense(id);        break;    case "update":        var updateIdStr = GetArgValue("--id");        var updateDesc = GetArgValue("--description");        var updateAmountStr = GetArgValue("--amount");        if (int.TryParse(updateIdStr, out int uId))        {            decimal? uAmount = decimal.TryParse(updateAmountStr, out decimal val) ? val : null;            service.UpdateExpense(uId, updateDesc, uAmount);        }        else            Console.WriteLine("Please provide a valid ID for update: --id [number]");        break;    case "export":        service.ExportToCsv();        break;    default:        Console.WriteLine($"Command '{command}' not recognized.");        break;}
